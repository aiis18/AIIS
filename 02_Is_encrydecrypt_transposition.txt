import math

def get_column_index(key, value):
    return key.index(str(value))


def encrypt(text, key):
    cols = len(key)
    rows = math.ceil(len(text) / cols)

    matrix = [['' for _ in range(cols)] for _ in range(rows)]

    index = 0

    for i in range(rows):
        for j in range(cols):
            if index < len(text):
                matrix[i][j] = text[index]
                index += 1

    cipher_text = ""

    for k in range(1, cols + 1):
        col = get_column_index(key, k)

        for i in range(rows):
            if matrix[i][col] != '':
                cipher_text += matrix[i][col]

    return cipher_text


def decrypt(cipher_text, key):
    cols = len(key)
    rows = math.ceil(len(cipher_text) / cols)

    matrix = [['' for _ in range(cols)] for _ in range(rows)]

    index = 0

    for k in range(1, cols + 1):
        col = get_column_index(key, k)

        for i in range(rows):
            if index < len(cipher_text):
                matrix[i][col] = cipher_text[index]
                index += 1

    plain_text = ""

    for i in range(rows):
        for j in range(cols):
            if matrix[i][j] != '':
                plain_text += matrix[i][j]

    return plain_text


plain_text = input("Enter Plain Text: ").replace(" ", "").upper()
key = input("Enter Key: ")

cipher_text = encrypt(plain_text, key)

print("\nCipher Text:", cipher_text)
print("Decrypted Text:", decrypt(cipher_text, key))
